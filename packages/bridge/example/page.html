📊 新旧架构对比

旧方案

// 字符串路径，无类型安全
await bridge.call('management.getAll')

// 需要手动处理事件
bridge.on('management.onInstalled', handler)
bridge.off('management.onInstalled', handler)

新方案 ✨

// 完全类型安全，自动补全
await bridge.extensions.getAll.query()

// 自动清理
const unsub = bridge.extensions
  .onChanged.subscribe(handler)
unsub() // 自动清理
